264 QString MusicAdvancedSearchedWidget::getSearchedKeyWork(int type, const QString &url)
265 {
266  QString key;
267  switch(M_SETTING_PTR->value(MusicSettingManager::DownloadServerChoiced).toInt())
268  {
269  case 0:
270  {
271  QRegExp regx("id=(\\d+)");
272  key = (url.indexOf(regx) != -1) ? regx.cap(1) : url;
273  break;
274  }
275  case 1:
276  {
277  QRegExp regx("/(\\w+).html");
278  key = (url.indexOf(regx) != -1) ? regx.cap(1) : url;
279  break;
280  }
281  case 2:
282  {
283  QRegExp regx(type == 4 ? "/mv/(\\w+)$" : "/(\\d+)");
284  key = (url.indexOf(regx) != -1) ? regx.cap(1) : url;
285  break;
286  }
287  case 3:
288  {
289  QRegExp regx("/(\\d+)");
290  key = (url.indexOf(regx) != -1) ? regx.cap(1) : url;
291  break;
292  }
293  case 4:
294  {
295  QRegExp regx(type == 3 ? "pid=(\\d+)" : "/(\\d+)");
296  key = (url.indexOf(regx) != -1) ? regx.cap(1) : url;
297  break;
298  }
299  case 5:
300  {
301  QRegExp regx;
302  if(type == 0)
303  regx.setPattern("hash=(\\w+)");
304  else if(type == 4)
305  regx.setPattern("/mv_(\\d+)");
306  else
307  regx.setPattern("/(\\d+)");
308  key = (url.indexOf(regx) != -1) ? regx.cap(1) : url;
309  break;
310  }
311  case 6:
312  {
313  QRegExp regx;
314  if(type == 0)
315  regx.setPattern("200000000(\\w+).html");
316  else if(type == 3)
317  regx.setPattern("/(\\w+).html");
318  else if(type == 4)
319  regx.setPattern("=(\\w+)");
320  else
321  regx.setPattern("/(\\d+)");
322  key = (url.indexOf(regx) != -1) ? regx.cap(1) : url;
323  break;
324  }
325  default: break;
326  }
327 
328  return key;
329 }
