904  if(idx != -1)

927  m_ui->musicPlayedList->remove(index[i]);
928  if(i != 0 && !contains && oldIndex <= index[i] && oldIndex >= index[i - 1])
929  {
930  oldIndex -= i;
931  contains = true;
932  }
933  }
934 
935  if(!contains && m_musicPlaylist->currentIndex() > index[0])
936  {
937  oldIndex -= index.count();
938  }
939  if(oldIndex == m_musicPlaylist->mediaCount())
940  {
941  --oldIndex;
942  }
943  m_musicPlaylist->setCurrentIndex(oldIndex);
944 
945  if(contains)
946  {
947  //The corresponding item is deleted from the Playlist
948  m_playControl = true;

950  m_playControl = false;

955  }
956  }
957  }
