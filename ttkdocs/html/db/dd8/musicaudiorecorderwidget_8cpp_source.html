239  len = 4096;

241  //Read sound samples from input device to buffer
242  const qint64 l = m_mpInputDevSound->read(m_mBuffer.data(), len);
243  if(l > 0)
244  {
245  //Assign sound samples to short array
246  short* resultingData = (short*)m_mBuffer.data();
247  short *outdata=resultingData;
248  outdata[ 0 ] = resultingData [ 0 ];
249  int iIndex;
250  if(false)
251  {

253  for ( iIndex=1; iIndex < len; iIndex++ )
254  {
255  outdata[ iIndex ] = 0.333 * resultingData[iIndex ] + ( 1.0 - 0.333 ) * outdata[ iIndex-1 ];
